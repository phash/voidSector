## Migration Pattern (reference for all tasks)

### Component migration

```tsx
// 1. Add import at top
import { useTranslation } from 'react-i18next';

// 2. Inside component function, first line:
const { t } = useTranslation('ui');

// 3. Replace strings:
// Simple:     "KEIN SCHIFF"  →  t('ship.noShip')
// With data:  `BUDGET: ${n}/100 XP`  →  t('ship.budget', { current: n, max: 100 })

// 4. For btn() usage:
import { btn } from '../ui-helpers';  // instead of '../ui-strings'
// btn(UI.actions.JUMP)  →  btn(t('actions.jump'))
```

### Test migration

With the mock in setup.ts, `t('ship.noShip')` returns `'ship.noShip'` in tests.

```tsx
// Before:
expect(screen.getByText('KEIN SCHIFF')).toBeInTheDocument();
// After:
expect(screen.getByText('ship.noShip')).toBeInTheDocument();

// Before (btn):
expect(screen.getByText('[JUMP]')).toBeInTheDocument();
// After (btn wraps the key):
expect(screen.getByText('[actions.jump]')).toBeInTheDocument();
```

```ts
/** Wraps a string in CRT bracket notation: ACCEPT → [ACCEPT] */
export function btn(label: string): string {
  return `[${label}]`;
}

/** Wraps a disabled button label with its blocking reason: JUMP → [JUMP — NO AP] */
export function btnDisabled(label: string, reason: string): string {
  return `[${label} — ${reason}]`;
}
```

```ts
import i18n from 'i18next';
import { initReactI18next } from 'react-i18next';
import LanguageDetector from 'i18next-browser-languagedetector';
import de from './locales/de/ui.json';
import en from './locales/en/ui.json';

i18n
  .use(LanguageDetector)
  .use(initReactI18next)
  .init({
    resources: {
      de: { ui: de },
      en: { ui: en },
    },
    fallbackLng: false,
    load: 'languageOnly',
    defaultNS: 'ui',
    ns: ['ui'],
    interpolation: { escapeValue: false },
    detection: {
      order: ['localStorage', 'navigator'],
      lookupLocalStorage: 'vs_language',
      caches: ['localStorage'],
    },
  });

export default i18n;
```

```ts
      if (opts) {

          (s, [k, v]) => s.replace(`{{${k}}}`, String(v)),
```
