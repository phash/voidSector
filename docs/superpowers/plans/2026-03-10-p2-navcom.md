# P2-NavCom Implementation Plan

> **For agentic workers:** REQUIRED: Use superpowers:subagent-driven-development (if subagents available) or superpowers:executing-plans to implement this plan. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Ersetze StationTerminalOverlay durch NavCom-zentrische Interaktion: Sektoren auf dem Radar anklicken → Objekte in Fenster 3 → Interaktion als Sub-Mode in Fenster 2.

**Architecture:** Neues `navcomMode`/`navcomTarget` in uiSlice steuert was CockpitLayout in Sec 2 rendert wenn NAV-COM aktiv ist. TradeScreen/MiningScreen/QuestsScreen bekommen optionale props für den navcom-Kontext-Header. Neue Komponenten `NavComHub` (Station/Jumpgate-Actions) und `NavComSectorDetail` (Fenster 3 Objekt-Liste). StationTerminalOverlay und TRADE-Programm werden vollständig entfernt.

**Tech Stack:** React 18, Zustand (zustand slice pattern), TypeScript strict, Vitest + RTL, `@void-sector/shared` constants

```typescript
// packages/client/src/components/NavComSectorDetail.tsx
import { useStore } from '../state/store';
import type { NavcomTarget } from '../state/uiSlice';

function getObjectsFromSector(sector: any, isLocal: boolean, onlinePlayers: any[]) {
  const objects: Array<{ icon: string; label: string; type: NavcomTarget['type'] | 'player' | 'npc'; data: any }> = [];

  if (!sector) return objects;

  if (sector.type === 'asteroid' || sector.type === 'nebula') {
    objects.push({
      icon: '⬡',
      label: `${sector.type.toUpperCase()} FIELD`,
      type: sector.type === 'asteroid' ? 'asteroid' : 'nebula',
      data: sector,
    });
  }
  if (sector.type === 'station') {
    objects.push({
      icon: '◈',
      label: sector.stationName ?? 'STATION',
      type: 'station',
      data: sector,
    });
  }
  if (sector.hasJumpgate || sector.type === 'jumpgate') {
    objects.push({
      icon: '⟁',
      label: `JUMPGATE`,
      type: 'jumpgate',
      data: sector,
    });
  }
  if (isLocal && sector.npcs?.length > 0) {
    sector.npcs.forEach((npc: any) => {
      objects.push({ icon: '⚔', label: `NPC ${npc.type?.toUpperCase() ?? 'UNKNOWN'}`, type: 'npc', data: npc });
    });
  }

  onlinePlayers
    .filter((p) => p.x === sector.x && p.y === sector.y)
    .forEach((p) => {
      objects.push({ icon: '●', label: p.name, type: 'player', data: p });
    });

  return objects;
}

export function NavComSectorDetail() {
  const position = useStore((s) => s.position);
  const currentSector = useStore((s) => s.currentSector);
  const selectedSector = useStore((s) => s.selectedSector);   // { x, y } | null — set by radar click
  const discoveries = useStore((s) => s.discoveries);          // Record<string, SectorData> — full sector cache
  const onlinePlayers = useStore((s) => s.onlinePlayers ?? []);
  const openNavcomTarget = useStore((s) => s.openNavcomTarget);
  const setActiveProgram = useStore((s) => s.setActiveProgram);

  // Resolve which sector to display:
  // 1. If player is at selectedSector → use currentSector (full data)
  // 2. If selectedSector differs → look up in discoveries cache
  // 3. Fallback: currentSector
  const sector = (() => {
    if (!selectedSector) return currentSector;
    const isCurrentSector = position?.x === selectedSector.x && position?.y === selectedSector.y;
    if (isCurrentSector) return currentSector;
    const key = `${selectedSector.x}:${selectedSector.y}`;
    return discoveries?.[key] ?? currentSector;
  })();
  if (!sector) {
    return <div style={{ padding: '12px', color: '#555', fontFamily: 'monospace', fontSize: '0.75rem' }}>NO SECTOR SELECTED</div>;
  }

  const isLocal = !!(position && sector && position.x === sector.x && position.y === sector.y);
  const objects = getObjectsFromSector(sector, isLocal, onlinePlayers);

  function handleObjectClick(obj: (typeof objects)[number]) {
    if (obj.type === 'player') {
      setActiveProgram('COMMS');
      return;
    }
    if (!isLocal) return;
    if (obj.type === 'npc') return; // combat flow handled elsewhere
    const mode = obj.type === 'station' ? 'hub' : obj.type === 'jumpgate' ? 'hub' : 'mining';
    openNavcomTarget({ type: obj.type as NavcomTarget['type'], x: sector.x, y: sector.y, name: obj.label }, mode);
  }

  return (
    <div style={{ fontFamily: 'monospace', fontSize: '0.75rem', padding: '10px', height: '100%', overflowY: 'auto' }}>
      <div style={{ borderBottom: '1px solid #222', paddingBottom: '4px', marginBottom: '6px' }}>
        <span style={{ color: 'var(--color-primary)' }}>◈ SEKTOR ({sector.x}, {sector.y}) · {sector.type?.toUpperCase()}</span>
        {isLocal && <span style={{ color: '#0a0', marginLeft: '8px', fontSize: '0.65rem' }}>● HIER</span>}
      </div>

      {sector.resources?.length > 0 && (
        <div style={{ color: '#555', fontSize: '0.65rem', marginBottom: '6px' }}>
          RESSOURCEN: {sector.resources.map((r: string) => r.toUpperCase()).join(' · ')}
        </div>
      )}

      <div style={{ color: '#666', fontSize: '0.65rem', letterSpacing: '1px', marginBottom: '4px' }}>OBJEKTE</div>

      {objects.length === 0 && (
        <div style={{ color: '#333' }}>—</div>
      )}

      {objects.map((obj, i) => {
        const isClickable = isLocal && obj.type !== 'npc' || obj.type === 'player';
        return (
          <div
            key={i}
            style={{ display: 'flex', justifyContent: 'space-between', alignItems: 'center', padding: '3px 0', borderBottom: '1px solid #111', cursor: isClickable ? 'pointer' : 'default' }}
            onClick={isClickable ? () => handleObjectClick(obj) : undefined}
          >
            <span style={{ color: isLocal && obj.type !== 'npc' ? 'var(--color-primary)' : '#aaa' }}>
              {obj.icon} {obj.label}{isLocal && obj.type !== 'npc' && obj.type !== 'player' ? ' ›' : ''}
            </span>
            {obj.type === 'player' && (
              <button
                onClick={(e) => { e.stopPropagation(); handleObjectClick(obj); }}
                style={{ border: '1px solid #0a0', background: 'none', color: '#0a0', fontFamily: 'monospace', fontSize: '0.7rem', cursor: 'pointer', padding: '1px 4px' }}
              >
                [CHAT]
              </button>
            )}
          </div>
        );
      })}

      {!isLocal && (
        <div style={{ color: '#555', fontSize: '0.65rem', marginTop: '8px' }}>
          Navigiere hierher um zu interagieren.
        </div>
      )}
    </div>
  );
}
```
